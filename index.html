async def handle_webapp_data(update: Update, context: ContextTypes.DEFAULT_TYPE): """Обработка данных из Web App""" user_id = update.effective_user.id logger.info(f"=== ПОЛУЧЕНЫ ДАННЫЕ ОТ WEB APP от пользователя {user_id} ===") if not update.message.web_app_data: logger.error("Нет web_app_data в сообщении") await update.message.reply_text("❌ Ошибка: данные не получены") return data = update.message.web_app_data.data logger.info(f"Длина данных: {len(data)}") if not data: logger.error("Пустые данные") await update.message.reply_text("❌ Ошибка: получены пустые данные") return # Проверяем, что это изображение (начинается с data:image) if data.startswith('data:image'): logger.info("Получено изображение подписи!") try: import base64 from io import BytesIO from PIL import Image # Извлекаем base64 данные if ',' in data: image_data = data.split(',')[1] else: image_data = data image_bytes = base64.b64decode(image_data) logger.info(f"Размер изображения: {len(image_bytes)} байт") # Сохраняем изображение with open(f"signature_{user_id}.png", "wb") as f: f.write(image_bytes) # Вставляем в Excel (если нужно) try: from openpyxl import load_workbook from openpyxl.drawing.image import Image as XLImage wb = load_workbook("template.xlsx") ws = wb.active img = XLImage(BytesIO(image_bytes)) img.width = 150 img.height = 75 ws.add_image(img, "F49") wb.save("template.xlsx") await update.message.reply_text("✅ Подпись добавлена в Excel (ячейка F49)!") except Exception as e: logger.error(f"Ошибка Excel: {e}") await update.message.reply_text(f"✅ Изображение сохранено как signature_{user_id}.png") except Exception as e: logger.error(f"Ошибка обработки: {e}") await update.message.reply_text(f"❌ Ошибка обработки: {e}") else: logger.info(f"Получены текстовые данные: {data[:100]}") await update.message.reply_text(f"✅ Получены данные: {data[:100]}...")
